Backend Architecture (Node.js + Express.js)

        
const express = require("express");
const fetch = require("node-fetch");
const app = express();

app.get("/api/routes", async (req, res) => {
  const { from, to, date } = req.query;

  try {
    const trainData = await fetch(`https://api.irctc.co.in/search?from=${from}&to=${to}&date=${date}`).then(r => r.json());
    const busData = await fetch(`https://api.redbus.in/search?from=${from}&to=${to}&date=${date}`).then(r => r.json());
    const cabData = await fetch(`https://api.uber.com/estimate?from=${from}&to=${to}`).then(r => r.json());

    res.json({
      trains: trainData,
      buses: busData,
      cabs: cabData
    });
  } catch (error) {
    res.status(500).json({ error: "API fetch failed" });
  }
});

app.listen(5000, () => console.log("Server running on port 5000"));
